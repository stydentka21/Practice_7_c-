#include <iostream>
#include<fstream>
using namespace std;
int main()
{
	ifstream in("ourfile.txt");
		if (!in.is_open()) {
		cout << "an error has occurred";
	}
	else
	{
		int i = 0;
		int length = in.tellg();
		char* buffer = new char[length];
		while (!in.eof())
		{

			char ch;
			in.get(ch);
			//cout << ch;
			buffer[i] = ch;
			i += 1;
		}
		in.close();
		cout << buffer;
	}
return 0;
}
